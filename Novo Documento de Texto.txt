function validarLuhn(numero) {
    if (numero.length !== 16) return false;
    if (/^0+$/.test(numero)) return false; 
    
    let soma = 0;
    let alternar = false;
    
    for (let i = numero.length - 1; i >= 0; i--) {
        let n = parseInt(numero.charAt(i), 10);
        if (alternar) {
            n *= 2;
            if (n > 9) n -= 9;
        }
        soma += n;
        alternar = !alternar;
    }
    return (soma % 10 === 0);
}

function validarData(dataString) {
    if (dataString.length !== 5) return false;
    
    const partes = dataString.split('/');
    if (partes.length !== 2) return false;

    const mesInserido = parseInt(partes[0], 10);
    const anoInserido = parseInt("20" + partes[1], 10); 

    if (mesInserido < 1 || mesInserido > 12) return false;

    const dataAtual = new Date();
    const mesAtual = dataAtual.getMonth() + 1; 
    const anoAtual = dataAtual.getFullYear();

    if (anoInserido < anoAtual) return false;
    if (anoInserido === anoAtual && mesInserido < mesAtual) return false;

    return true;
}

    } else {

function validarLuhn(numero) {
    if (numero.length !== 16) return false;
    if (/^0+$/.test(numero)) return false; 
    
    let soma = 0;
    let alternar = false;
    
    for (let i = numero.length - 1; i >= 0; i--) {
        let n = parseInt(numero.charAt(i), 10);
        if (alternar) {
            n *= 2;
            if (n > 9) n -= 9;
        }
        soma += n;
        alternar = !alternar;
    }
    return (soma % 10 === 0);
}

function validarData(dataString) {
    if (dataString.length !== 5) return false;
    
    const partes = dataString.split('/');
    if (partes.length !== 2) return false;

    const mesInserido = parseInt(partes[0], 10);
    const anoInserido = parseInt("20" + partes[1], 10); 

    if (mesInserido < 1 || mesInserido > 12) return false;

    const dataAtual = new Date();
    const mesAtual = dataAtual.getMonth() + 1; 
    const anoAtual = dataAtual.getFullYear();

    if (anoInserido < anoAtual) return false;
    if (anoInserido === anoAtual && mesInserido < mesAtual) return false;

    return true;
}
